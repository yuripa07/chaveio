# i18n (pt-BR / English) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add pt-BR / English language selection with browser-language detection and localStorage persistence, translating all UI strings across pages and components.

**Architecture:** A single `src/locales/translations.ts` file holds the `Translations` interface and both locale objects. A `LocaleProvider` client component (React context) wraps the app in `layout.tsx` and exposes a `useLocale()` hook returning `{ t, locale, setLocale }`. A floating `LocaleSwitcher` component allows toggling. Every client page and component that has hardcoded strings imports `useLocale()`.

**Tech Stack:** React Context API, localStorage, `navigator.language`, TypeScript (typed translation object), no external i18n library.

// src/contexts/locale-context.tsx
"use client";

import { createContext, useCallback, useContext, useEffect, useState } from "react";
import { type Locale, type Translations, translations } from "@/locales/translations";

const STORAGE_KEY = "chaveio:locale";

function detectLocale(): Locale {
  try {
    const stored = localStorage.getItem(STORAGE_KEY);
    if (stored === "pt-BR" || stored === "en") return stored;
  } catch {}
  const lang = navigator.language ?? "";
  return lang.startsWith("pt") ? "pt-BR" : "en";
}

type LocaleContextValue = {
  locale: Locale;
  t: Translations;
  setLocale: (locale: Locale) => void;
};

const LocaleContext = createContext<LocaleContextValue>({
  locale: "pt-BR",
  t: translations["pt-BR"],
  setLocale: () => {},
});

export function LocaleProvider({ children }: { children: React.ReactNode }) {
  const [locale, setLocaleState] = useState<Locale>("pt-BR");

  useEffect(() => {
    const detected = detectLocale();
    setLocaleState(detected);
  }, []);

  useEffect(() => {
    document.documentElement.lang = locale;
  }, [locale]);

  const setLocale = useCallback((next: Locale) => {
    setLocaleState(next);
    try {
      localStorage.setItem(STORAGE_KEY, next);
    } catch {}
  }, []);

  return (
    <LocaleContext.Provider value={{ locale, t: translations[locale], setLocale }}>
      {children}
    </LocaleContext.Provider>
  );
}

export function useLocale(): LocaleContextValue {
  return useContext(LocaleContext);
}
